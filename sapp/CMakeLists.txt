if (FIPS_EMSCRIPTEN)
    add_definitions(-DSOKOL_GLES2)
elseif (FIPS_RASPBERRYPI)
    add_definitions(-DSOKOL_GLES2)
elseif (FIPS_ANDROID)
    add_definitions(-DSOKOL_GLES3)
elseif (SOKOL_USE_D3D11)
    add_definitions(-DSOKOL_D3D11)
elseif (SOKOL_USE_METAL)
    add_definitions(-DSOKOL_METAL)
else()
    if (FIPS_IOS)
        add_definitions(-DSOKOL_GLES3)
    else()
        add_definitions(-DSOKOL_GLCORE33)
    endif()
endif()

if (FIPS_WINDOWS)
    fips_begin_lib(gl)
        fips_files(gl.c gl.h)
    fips_end_lib()
endif()

fips_begin_lib(sokol)
    fips_vs_warning_level(3)
    if (FIPS_OSX)
        fips_files(sokol.m)
        if (FIPS_IOS)
            fips_frameworks_osx(UIKit)
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(Metal MetalKit)
            else()
                fips_frameworks_osx(OpenGLES GLKit)
            endif()
        else()
            fips_frameworks_osx(Cocoa QuartzCore)
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(Metal MetalKit)
            else()
                fips_frameworks_osx(OpenGL)
            endif()
        endif()
    else()
        fips_files(sokol.c)
    endif()
    if (FIPS_WINDOWS)
        fips_deps(gl)
    endif()
fips_end_lib()

fips_begin_app(clear-sapp windowed)
    fips_vs_warning_level(3)
    fips_files(clear-sapp.c)
    fips_deps(sokol)
fips_end_app()

fips_begin_app(cube-sapp windowed)
    fips_vs_warning_level(3)
    fips_files(cube-sapp.c)
    fips_deps(sokol)
fips_end_app()

fips_begin_app(imgui-sapp windowed)
    fips_vs_warning_level(3)
    fips_files(imgui-sapp.cc)
    fips_deps(sokol imgui)
fips_end_app()
